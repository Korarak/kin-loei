import { renderProfile } from './pages/profile.js'
import { renderScan } from './pages/scan.js'
import { renderResult } from './pages/result.js'
import { renderHistory } from './pages/history.js'

const routes = {
  '#/profile': renderProfile,
  '#/scan':    renderScan,
  '#/result':  renderResult,
  '#/history': renderHistory,
}

function navigate() {
  const hash = location.hash || '#/scan'
  const render = routes[hash] ?? renderScan
  document.getElementById('app').innerHTML = ''
  render(document.getElementById('app'))
  // update active tab
  document.querySelectorAll('.tab-btn').forEach(b =>
    b.classList.toggle('active', b.dataset.href === hash))
}

window.addEventListener('hashchange', navigate)

jssrc/camera.js
export async function startCamera(videoEl) {
  const stream = await navigator.mediaDevices.getUserMedia({
    video: { facingMode: 'environment' }, // กล้องหลัง
    audio: false,
  })
  videoEl.srcObject = stream
  return stream
}

export function captureFrame(videoEl) {
  const canvas = document.createElement('canvas')
  canvas.width  = videoEl.videoWidth
  canvas.height = videoEl.videoHeight
  canvas.getContext('2d').drawImage(videoEl, 0, 0)
  return new Promise(res => canvas.toBlob(res, 'image/jpeg', 0.85))
}

export function stopCamera(stream) {
  stream?.getTracks().forEach(t => t.stop())
}

const BASE = import.meta.env.VITE_API_URL ?? 'http://localhost:8000'

export async function scanFood({ deviceId, profile, imageBlob, text }) {
  const form = new FormData()
  form.append('device_id', deviceId)
  form.append('health_profile', JSON.stringify(profile))
  if (imageBlob) form.append('image', imageBlob, 'scan.jpg')
  if (text)      form.append('text_input', text)

  const res = await fetch(`${BASE}/analyze/scan`, { method: 'POST', body: form })
  if (!res.ok) throw new Error(await res.text())
  return res.json()
}

export async function getHistory(deviceId) {
  const res = await fetch(`${BASE}/analyze/history/${deviceId}`)
  return res.json()
}

  (await db).get('profile', 'health') ?? {}

export const saveScan = async (result) =>
  (await db).add('scans', { ...result, savedAt: Date.now() })

export const getScans = async () =>
  (await db).getAll('scans')

def create_token(sub, role="user", hours=168):  # 7 วัน
    payload = {"sub": sub, "role": role, "exp": datetime.utcnow() + timedelta(hours=hours)}
    return jwt.encode(payload, settings.secret_key, algorithm="HS256")

Docker Compose · เพิ่ม Admin Service

เพิ่มใน docker-compose.yml

yamldocker-compose.yml — เพิ่ม service admin
  # ── 4. Admin Dashboard ──────────────────────────
  admin:
    build:
      context: ./admin
      dockerfile: Dockerfile
    container_name: kinloei-admin
    restart: unless-stopped
    ports:
      - "${ADMIN_PORT:-18081}:80"
    networks:
      - app-net
    depends_on:
      backend:
        condition: service_healthy
    labels:
      com.kinloei.service: admin
      com.kinloei.description: "Admin Dashboard — Vite + Tailwind CSS"

Step 4.2
JWT Auth
~4 ชั่วโมง

Login + JWT + Auth Guard

Backend สร้าง routers/admin.py และ core/admin_auth.py — Frontend login form + ดัก route ที่ยังไม่ login

pythonbackend/core/admin_auth.py
from jose import jwt from fastapi import HTTPException, Depends from fastapi.security import HTTPBearer def create_admin_token(settings) -> str: return jwt.encode( {"sub": "admin", "exp": datetime.utcnow() + timedelta(hours=8)}, settings.admin_jwt_secret, algorithm="HS256" ) security = HTTPBearer() async def verify_admin_token( credentials = Depends(security), settings: Settings = Depends(get_settings) ): try: jwt.decode(credentials.credentials, settings.admin_jwt_secret, ["HS256"]) except: raise HTTPException(401, "Invalid admin token")

FastAPI Endpoints

เหมือนเดิมทุกอย่าง — PWA เรียกผ่าน fetch() แทน dio เท่านั้น

MethodEndpointPWA ส่งอะไรได้คืน
GET /health — {"status":"ok"}
POST /analyze/scan FormData: device_id, health_profile (JSON string), image (Blob จาก canvas.toBlob), text_input scan_id + result: status, ingredients, flagged_ingredients, summary, disclaimer
GET /analyze/history/{device_id} device_id (path param) array of scan history (backup จาก server)
GET /profile/{device_id} device_id health_profile JSON
PUT /profile/{device_id} JSON body: conditions, allergies, avoid_ingredients updated profile
DELETE /profile/{device_id} device_id ลบข้อมูลทั้งหมด (PDPA compliance)
POST /auth/register JSON: email, password, display_name?, device_id? JWT token + user info · link device ที่มีอยู่กับบัญชีใหม่ได้
POST /auth/login JSON: email, password JWT token (7 วัน) + user info
GET /auth/me Bearer token (header) ข้อมูล user ปัจจุบัน
POST /admin/login JSON: username, password (env credentials) JWT token role=admin (8 ชม.)
GET /admin/users Bearer (admin) · ?page &size &search paginated user list
GET /admin/stats Bearer (admin) users total/registered · scans SAFE/CAUTION/AVOID count
